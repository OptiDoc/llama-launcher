     1→/**
     2→ * LlamaLauncher state store + simulator.
     3→ *
     4→ * This is a client-side simulation of llama.cpp server instances.
     5→ * Each "instance" represents a llama-server process that streams
     6→ * realistic startup logs to its own console buffer.
     7→ *
     8→ * Extended with:
     9→ * - Workspaces (isolated environments)
    10→ * - App status / hibernation (auto-unload models after idle period,
    11→ *   hot-reload on next request)
    12→ * - Profiles with scope: 'global' | 'model' (for sharing + auto-calibration)
    13→ * - HuggingFace model download queue with quantization picker
    14→ * - Real-time system metrics stream (CPU/RAM/GPU/tok/s) for the dashboard
    15→ */
    16→

   140→function uid(prefix = "id") {

   144→function nowTs() {
   145→  return Date.now();

   148→function emitLog(instanceId: string, kind: LogKind, text: string) {
   149→  const line: ConsoleLine = {
   150→    id: uid("log"),
   151→    instanceId,
   152→    ts: nowTs(),
   153→    kind,
   154→    text,
   155→  };
   156→  logSubscribers.forEach((fn) => fn(line));
   157→  return line;
   158→}

   164→function runStartupSequence(
   165→  instance: LlamaInstance,
   166→  store: LlamaStore,
   167→) {
   168→  let cancelled = false;
   169→  let reqTimer: ReturnType<typeof setTimeout> | null = null;
   170→
   171→  const sim: SimState = {
   172→    running: true,
   173→    ticks: 0,
   174→    stop: () => {
   175→      cancelled = true;
   176→      sim.running = false;
   177→      if (reqTimer) clearTimeout(reqTimer);
   178→    },
   179→  };
   180→  instanceSims.set(instance.id, sim);
   181→
   182→  const steps: Array<[number, LogKind, string]> = [
   183→    [50, "info", `$ llama-server --model ${instance.model} --host ${instance.host} --port ${instance.port} -c ${instance.ctxSize} -t ${instance.threads} -ngl ${instance.gpu === "cpu" ? 0 : 99}`],
   184→    [120, "debug", "build: 4402 (8f1f7e1) with AVX2=1 AVX512=1 BMI2=1 CUDA=1"],
   185→    [180, "info", "system_info: n_threads = " + instance.threads + " n_gpu_layers = 99"],
   186→    [260, "info", "loading model from " + instance.model],
   187→    [380, "debug", "llama_model_loader: loaded meta data with 24 KV pairs"],
   188→    [440, "debug", "llama_model_loader: - kv[ 0]:                       general.name = " + instance.name],
   189→    [520, "debug", "llama_model_loader: - kv[ 1]:          general.architecture = llama"],
   190→    [600, "info", `llama_model_loader: - kv[12]:                      llama.context_length = ${instance.ctxSize}`],
   191→    [680, "debug", "llama_model_loader: - type  f16:  221 tensors"],
   192→    [760, "info", "llama_model_loader: loading model part 1/1 from '" + instance.model + "'"],
   193→    [900, "info", "llama_model_loader: model size = " + (instance.model.includes("7b") ? "13.9" : "4.1") + " GiB"],
   194→    [980, "success", "llama_model_loader: model loaded successfully"],
   195→    [1060, "info", "using CUDA for GPU acceleration"],
   196→    [1140, "debug", "llama_kv_cache:  CUDA0 KV buffer size = " + Math.round(instance.ctxSize * 0.5) + " MiB"],
   197→    [1220, "info", "llama_new_context_with_model: n_ctx = " + instance.ctxSize + ", batch = 512"],
   198→    [1320, "success", "llama_new_context_with_model: graph initialized"],
   199→    [1400, "info", `llama server: listening on ${instance.host}:${instance.port}`],
   200→    [1480, "info", "llama server: loading model took " + (1200 + Math.floor(Math.random() * 600)) + " ms"],
   201→    [1560, "success", `llama server: model loaded, ready for requests at http://${instance.host}:${instance.port}`],
   202→  ];
   203→
   204→  let elapsed = 0;
   205→  steps.forEach(([delay, kind, text]) => {
   206→    if (cancelled) return;
   207→    setTimeout(() => {
   208→      if (cancelled) return;
   209→      emitLog(instance.id, kind, `[${fmtTime(new Date())}] ${text}`);
   210→    }, delay);
   211→    elapsed = delay;
   212→  });
   213→
   214→  setTimeout(() => {
   215→    if (cancelled) return;
   216→    store.markRunning(instance.id);
   217→    emitLog(instance.id, "success", `[${fmtTime(new Date())}] ✓ Server ready. Listening on http://${instance.host}:${instance.port}/completions`);
   218→
   219→    const tick = () => {
   220→      if (cancelled || !sim.running) return;
   221→      sim.ticks += 1;
   222→      const t = sim.ticks;
   223→      // Each request bumps activity -> resets idle timer
   224→      store.registerActivity();
   225→      if (t % 3 === 0) {
   226→        const promptTok = 8 + Math.floor(Math.random() * 80);
   227→        const genTok = 12 + Math.floor(Math.random() * 120);
   228→        const tps = 18 + Math.random() * 24;
   229→        emitLog(instance.id, "info", `[${fmtTime(new Date())}] POST /v1/chat/completions 200 - prompt: ${promptTok} tok, gen: ${genTok} tok, ${tps.toFixed(1)} tok/s`);
   230→        store.bumpStats(instance.id, promptTok, genTok, tps);
   231→      } else if (t % 5 === 0) {
   232→        emitLog(instance.id, "debug", `[${fmtTime(new Date())}] slot 0: cache reused, kv tokens = ${100 + Math.floor(Math.random() * 400)}`);
   233→      } else if (t % 7 === 0) {
   234→        emitLog(instance.id, "warn", `[${fmtTime(new Date())}] request queue depth = ${1 + Math.floor(Math.random() * 3)}`);
   235→      }
   236→      reqTimer = setTimeout(tick, 1800 + Math.random() * 1200);
   237→    };
   238→    reqTimer = setTimeout(tick, 1800);
   239→  }, elapsed + 200);
   240→}
   241→
   242→function runShutdownSequence(
   243→  instance: LlamaInstance,
   244→  store: LlamaStore,
   245→) {
   246→  const sim = instanceSims.get(instance.id);
   247→  if (sim) sim.stop();
   248→
   249→  emitLog(instance.id, "warn", `[${fmtTime(new Date())}] received SIGINT, shutting down...`);
   250→  setTimeout(() => {
   251→    emitLog(instance.id, "info", `[${fmtTime(new Date())}] waiting for in-flight requests to finish`);
   252→  }, 200);
   253→  setTimeout(() => {
   254→    emitLog(instance.id, "info", `[${fmtTime(new Date())}] freeing KV cache and model buffers`);
   255→  }, 450);
   256→  setTimeout(() => {
   257→    emitLog(instance.id, "success", `[${fmtTime(new Date())}] server stopped cleanly. goodbye.`);
   258→    store.markStopped(instance.id);
   259→  }, 700);
   260→}

   410→// Idle/hibernate watchdog
   411→let watchdogTimer: ReturnType<typeof setInterval> | null = null;
   412→
   413→function startWatchdog(store: LlamaStore) {
   414→  if (watchdogTimer) clearInterval(watchdogTimer);
   415→  watchdogTimer = setInterval(() => {
   416→    const s = useLlamaStore.getState();
   417→    const since = Date.now() - s.lastActivityAt;
   418→    const running = s.instances.filter((i) => i.status === "running" || i.status === "starting");
   419→
   420→    if (s.appStatus === "waking" || s.appStatus === "hibernating") return;
   421→
   422→    if (since >= IDLE_THRESHOLD_MS + HIBERNATE_THRESHOLD_MS && running.length > 0 && s.appStatus !== "hibernating") {
   423→      // Hibernate: unload all running models
   424→      s.setAppStatus("hibernating");
   425→      emitLog(SYSTEM_CONSOLE_ID, "warn", `[${fmtTime(new Date())}] [hibernate] idle ${Math.round(since / 1000)}s — unloading ${running.length} model(s) from VRAM`);
   426→      running.forEach((inst) => {
   427→        useLlamaStore.getState().stopInstance(inst.id);
   428→        useLlamaStore.setState((st) => ({
   429→          hibernatedInstanceIds: [...st.hibernatedInstanceIds, inst.id],
   430→          instances: st.instances.map((i) =>
   431→            i.id === inst.id
   432→              ? {
   433→                  ...i,
   434→                  hibernatedConfig: { name: i.name, model: i.model, profile: i.profile, port: i.port, host: i.host, gpu: i.gpu },
   435→                }
   436→              : i,
   437→          ),
   438→        }));
   439→      });
   440→      setTimeout(() => {
   441→        emitLog(SYSTEM_CONSOLE_ID, "info", `[${fmtTime(new Date())}] [hibernate] all models unloaded. VRAM freed. Awaiting next request to hot-reload.`);
   442→      }, 1200);
   443→    } else if (since >= IDLE_THRESHOLD_MS && s.appStatus === "active") {
   444→      s.setAppStatus("idle");
   445→      emitLog(SYSTEM_CONSOLE_ID, "debug", `[${fmtTime(new Date())}] [idle] no activity for ${Math.round(since / 1000)}s — will hibernate in ${Math.round(HIBERNATE_THRESHOLD_MS / 1000)}s`);
   446→    } else if (since < IDLE_THRESHOLD_MS && s.appStatus === "idle") {
   447→      s.setAppStatus("active");
   448→    }
   449→  }, 3000);
   450→}

   452→// Real-time metrics generator (1s tick)
   453→let metricsTimer: ReturnType<typeof setInterval> | null = null;
   454→function startMetricsTicker(store: LlamaStore) {
   455→  if (metricsTimer) clearInterval(metricsTimer);
   456→  metricsTimer = setInterval(() => {
   457→    const s = useLlamaStore.getState();
   458→    const running = s.instances.filter((i) => i.status === "running");
   459→    const totalTps = running.reduce((sum, i) => sum + i.tokensPerSec, 0);
   460→    const totalMem = running.reduce((sum, i) => sum + i.memoryMb, 0);
   461→    const totalReq = running.reduce((sum, i) => sum + i.requestsPerMin, 0);
   462→    const isHibernating = s.appStatus === "hibernating";
   463→
   464→    const sample: MetricSample = {
   465→      t: Date.now(),
   466→      cpu: isHibernating ? 1 + Math.random() * 2 : 8 + Math.random() * 18 + running.length * 3,
   467→      ram: isHibernating ? 18 + Math.random() * 4 : 32 + Math.random() * 6 + totalMem / 1024,
   468→      gpu: isHibernating ? 0 : Math.min(100, 15 + running.length * 22 + Math.random() * 18),
   469→      gpuMem: isHibernating ? 0 : Math.min(100, (totalMem / 12288) * 100),
   470→      tps: totalTps,
   471→      reqPerMin: totalReq,
   472→    };
   473→    store.pushMetric(sample);
   474→  }, 1500);
   475→}

   477→export const useLlamaStore = create<LlamaStore>((set, get) => {
   478→  logSubscribers.add((line) => {
   479→    get().appendLog(line);
   480→  });
   481→
   482→  const store: LlamaStore = {
   483→    instances: seedInstances,
   484→    models: seedModels,
   485→    profiles: seedProfiles,
   486→    releases: seedReleases,
   487→    workspaces: seedWorkspaces,
   488→    activeWorkspaceId: seedWorkspaces[0].id,
   489→    downloads: [],
   490→    logs: { [SYSTEM_CONSOLE_ID]: initialSystemLogs },
   491→    activeConsoleId: SYSTEM_CONSOLE_ID,
   492→    consoleOpen: false,
   493→    consoleHeight: 240,
   494→
   495→    appStatus: "active",
   496→    lastActivityAt: nowTs(),
   497→    hibernatedInstanceIds: [],
   498→    metrics: seedMetrics(),
   499→
   500→    registerActivity: () => {
   501→      const s = get();
   502→      const wasHibernating = s.appStatus === "hibernating";
   503→      set({ lastActivityAt: nowTs() });
   504→      if (wasHibernating) {
   505→        // Wake: hot-reload hibernated instances
   506→        set({ appStatus: "waking" });
   507→        emitLog(SYSTEM_CONSOLE_ID, "info", `[${fmtTime(new Date())}] [wake] activity detected — hot-reloading ${s.hibernatedInstanceIds.length} hibernated model(s)`);
   508→        const hibernatedIds = [...s.hibernatedInstanceIds];
   509→        set({ hibernatedInstanceIds: [] });
   510→        let delay = 0;
   511→        hibernatedIds.forEach((oldId) => {
   512→          const oldInst = s.instances.find((i) => i.id === oldId);
   513→          const cfg = oldInst?.hibernatedConfig;
   514→          if (!cfg) return;
   515→          setTimeout(() => {
   516→            // remove old stopped record, start fresh
   517→            get().removeInstance(oldId);
   518→            const newId = get().startInstance(cfg);
   519→            emitLog(newId, "info", `[${fmtTime(new Date())}] [wake] hot-reloaded from hibernation`);
   520→          }, delay);
   521→          delay += 400;
   522→        });
   523→        setTimeout(() => {
   524→          get().setAppStatus("active");
   525→          emitLog(SYSTEM_CONSOLE_ID, "success", `[${fmtTime(new Date())}] [wake] all models reloaded, resuming normal operation`);
   526→        }, delay + 2000);
   527→      } else if (s.appStatus === "idle") {
   528→        set({ appStatus: "active" });
   529→      }
   530→    },
   531→
   532→    setAppStatus: (st) => set({ appStatus: st }),
   533→    forceHibernate: () => {
   534→      set({ lastActivityAt: nowTs() - IDLE_THRESHOLD_MS - HIBERNATE_THRESHOLD_MS - 1000 });
   535→    },
   536→    forceWake: () => {
   537→      get().registerActivity();
   538→    },
   539→    pushMetric: (m) =>
   540→      set((s) => ({
   541→        metrics: [...s.metrics.slice(-59), m],
   542→      })),
   543→
   544→    setActiveWorkspace: (id) => {
   545→      set({ activeWorkspaceId: id });
   546→      const ws = get().workspaces.find((w) => w.id === id);
   547→      if (ws) emitLog(SYSTEM_CONSOLE_ID, "info", `[${fmtTime(new Date())}] [workspace] switched to "${ws.name}"`);
   548→    },
   549→    addWorkspace: (w) => set((s) => ({ workspaces: [...s.workspaces, { ...w, id: uid("ws") }] })),
   550→
   551→    startInstance: ({ name, model, profile, port, host, gpu }) => {
   552→      const id = uid("inst");
   553→      const prof = get().profiles.find((p) => p.id === profile) ?? get().profiles[0];
   554→      const colors: LlamaInstance["color"][] = ["green", "orange", "blue", "pink", "purple"];
   555→      const color = colors[get().instances.length % colors.length];
   556→      const instance: LlamaInstance = {
   557→        id,
   558→        name,
   559→        model,
   560→        profile: prof.name,
   561→        port,
   562→        host,
   563→        status: "starting",
   564→        gpu,
   565→        ctxSize: prof.ctxSize,
   566→        threads: prof.threads,
   567→        color,
   568→        startedAt: nowTs(),
   569→        promptTokens: 0,
   570→        generatedTokens: 0,
   571→        requestsPerMin: 0,
   572→        tokensPerSec: 0,
   573→        memoryMb: 0,
   574→      };
   575→      set((s) => ({
   576→        instances: [...s.instances, instance],
   577→        logs: { ...s.logs, [id]: [] },
   578→        activeConsoleId: id,
   579→        consoleOpen: true,
   580→        appStatus: "active",
   581→        lastActivityAt: nowTs(),
   582→      }));
   583→      emitLog(id, "info", `[${fmtTime(new Date())}] starting llama-server for "${name}" (model: ${model})`);
   584→      runStartupSequence(instance, get());
   585→      get().registerActivity();
   586→      return id;
   587→    },
   588→
   589→    stopInstance: (id) => {
   590→      const inst = get().instances.find((i) => i.id === id);
   591→      if (!inst) return;
   592→      set((s) => ({
   593→        instances: s.instances.map((i) => (i.id === id ? { ...i, status: "stopping" } : i)),
   594→      }));
   595→      runShutdownSequence(inst, get());
   596→    },
   597→
   598→    removeInstance: (id) => {
   599→      const sim = instanceSims.get(id);
   600→      if (sim) sim.stop();
   601→      instanceSims.delete(id);
   602→      set((s) => {
   603→        const newLogs = { ...s.logs };
   604→        delete newLogs[id];
   605→        const newInstances = s.instances.filter((i) => i.id !== id);
   606→        const newActive = s.activeConsoleId === id ? SYSTEM_CONSOLE_ID : s.activeConsoleId;
   607→        return { instances: newInstances, logs: newLogs, activeConsoleId: newActive };
   608→      });
   609→    },
   610→
   611→    markRunning: (id) =>
   612→      set((s) => ({
   613→        instances: s.instances.map((i) => (i.id === id ? { ...i, status: "running", startedAt: i.startedAt ?? nowTs() } : i)),
   614→      })),
   615→
   616→    markStopped: (id) =>
   617→      set((s) => ({
   618→        instances: s.instances.map((i) =>
   619→          i.id === id ? { ...i, status: "stopped", startedAt: undefined } : i,
   620→        ),
   621→      })),
   622→
   623→    bumpStats: (id, prompt, gen, tps) =>
   624→      set((s) => ({
   625→        instances: s.instances.map((i) =>
   626→          i.id === id
   627→            ? {
   628→                ...i,
   629→                promptTokens: i.promptTokens + prompt,
   630→                generatedTokens: i.generatedTokens + gen,
   631→                tokensPerSec: tps,
   632→                requestsPerMin: i.requestsPerMin + 1,
   633→                memoryMb: Math.round(i.ctxSize * 0.5 + 1200 + Math.random() * 200),
   634→              }
   635→            : i,
   636→        ),
   637→      })),
   638→
   639→    setActiveConsole: (id) => set({ activeConsoleId: id, consoleOpen: true }),
   640→    toggleConsole: () => set((s) => ({ consoleOpen: !s.consoleOpen })),
   641→    setConsoleOpen: (open) => set({ consoleOpen: open }),
   642→    setConsoleHeight: (h) => set({ consoleHeight: Math.max(120, Math.min(600, h)) }),
   643→    clearConsole: (id) => set((s) => ({ logs: { ...s.logs, [id]: [] } })),
   644→
   645→    downloadModel: (id) =>
   646→      set((s) => ({
   647→        models: s.models.map((m) => (m.id === id ? { ...m, downloaded: true } : m)),
   648→      })),
   649→
   650→    startHFDownload: ({ repo, quant, modelName }) => {
   651→      const dlId = uid("dl");
   652→      const q = HF_QUANTS.find((x) => x.id === quant);
   653→      const repoInfo = HF_POPULAR_REPOS.find((r) => r.repo === repo);
   654→      const sizeGb = (repoInfo?.baseSizeGb ?? 8) * (q?.sizeFactor ?? 0.6);
   655→      const filename = `${repo.split("/")[1]}-${quant}.gguf`;
   656→      const dl: HFDownload = {
   657→        id: dlId,
   658→        repo,
   659→        quant,
   660→        filename,
   661→        sizeGb,
   662→        progress: 0,
   663→        status: "downloading",
   664→        startedAt: nowTs(),
   665→        modelName,
   666→      };
   667→      set((s) => ({ downloads: [...s.downloads, dl] }));
   668→      emitLog(SYSTEM_CONSOLE_ID, "info", `[${fmtTime(new Date())}] [hf] downloading ${filename} from ${repo} (${sizeGb.toFixed(1)} GB)`);
   669→
   670→      // Simulate progressive download
   671→      let progress = 0;
   672→      const tick = () => {
   673→        progress += 2 + Math.random() * 6;
   674→        const s = useLlamaStore.getState();
   675→        if (progress >= 100) {
   676→          const newModel: LlamaModel = {
   677→            id: uid("m"),
   678→            name: `${modelName} ${quant}`,
   679→            family: repoInfo?.family ?? "unknown",
   680→            sizeGb: Math.round(sizeGb * 10) / 10,
   681→            quant,
   682→            downloaded: true,
   683→            path: `/models/${filename}`,
   684→            hfRepo: repo,
   685→          };
   686→          set((st) => ({
   687→            downloads: st.downloads.map((d) => (d.id === dlId ? { ...d, progress: 100, status: "completed" } : d)),
   688→            models: [...st.models, newModel],
   689→          }));
   690→          emitLog(SYSTEM_CONSOLE_ID, "success", `[${fmtTime(new Date())}] [hf] download complete: ${filename} (${sizeGb.toFixed(1)} GB)`);
   691→          return;
   692→        }
   693→        set((st) => ({
   694→          downloads: st.downloads.map((d) => (d.id === dlId ? { ...d, progress } : d)),
   695→        }));
   696→        setTimeout(tick, 350 + Math.random() * 250);
   697→      };
   698→      setTimeout(tick, 600);
   699→      return dlId;
   700→    },
   701→
   702→    installRelease: (id) =>
   703→      set((s) => ({
   704→        releases: s.releases.map((r) => (r.id === id ? { ...r, installed: true } : { ...r, installed: false })),
   705→      })),
   706→
   707→    addProfile: (p) => set((s) => ({ profiles: [...s.profiles, { ...p, id: uid("prof") }] })),
   708→
   709→    removeProfile: (id) =>
   710→      set((s) => ({ profiles: s.profiles.filter((p) => p.id !== id) })),
   711→
   712→    shareProfile: (id) =>
   713→      set((s) => ({
   714→        profiles: s.profiles.map((p) =>
   715→          p.id === id ? { ...p, shared: true, shareId: p.shareId ?? `sh_${id}_v1` } : p,
   716→        ),
   717→      })),
   718→    calibrateProfile: (id) =>
   719→      set((s) => ({
   720→        profiles: s.profiles.map((p) =>
   721→          p.id === id ? { ...p, calibrationScore: Math.min(100, (p.calibrationScore ?? 70) + 5 + Math.floor(Math.random() * 8)) } : p,

   723→      })),
   724→
   725→    appendLog: (line) =>
   726→      set((s) => {
   727→        const list = s.logs[line.instanceId] ?? [];
   728→        const next = list.length > 800 ? list.slice(list.length - 800) : list;
   729→        return { logs: { ...s.logs, [line.instanceId]: [...next, line] } };
   730→      }),
   731→  };
   732→
   733→  // Start background watchdog + metrics ticker (browser-only)
   734→  if (typeof window !== "undefined") {
   735→    setTimeout(() => {
   736→      startWatchdog(store);
   737→      startMetricsTicker(store);
   738→    }, 500);
   739→  }
   740→
   741→  return store;

   746→export function uptimeString(startedAt?: number) {
   747→  if (!startedAt) return "--";
   748→  const sec = Math.floor((Date.now() - startedAt) / 1000);
   749→  const h = Math.floor(sec / 3600);
   750→  const m = Math.floor((sec % 3600) / 60);
   751→  const s = sec % 60;
   752→  if (h > 0) return `${h}h ${m}m`;
   753→  if (m > 0) return `${m}m ${s}s`;
   754→  return `${s}s`;
   755→}
