     1→/**
     2→ * LlamaLauncher state store + simulator.
     3→ *
     4→ * Client-side simulation of llama.cpp server instances. Each instance is a
     5→ * llama-server process that streams realistic startup logs to its own console
     6→ * buffer.
     7→ *
     8→ * Features:
     9→ * - Workspaces (isolated environments with own instances/profiles/releases)
    10→ * - App status / hibernation (auto-unload models after idle period,
    11→ *   hot-reload on next request) — hibernate delay is configurable
    12→ * - Profiles with scope: 'global' | 'model' (for sharing + auto-calibration)
    13→ * - HuggingFace model download (search-first, quantization picker)
    14→ * - llama.cpp release variants (cuda12 / cuda13 / vulkan / cpu / …)
    15→ * - Notifications (new GitHub release, download complete, …)
    16→ * - Real-time system metrics stream (CPU/RAM/GPU/tok/s) for the dashboard
    17→ * - Model metadata: builder, architecture, license, edit/delete/missing
    18→ */

   263→export function searchHFModels(query: string): HFSearchResult[] {
   264→  const q = query.trim().toLowerCase();
   265→  if (!q) return [];
   266→  return HF_CATALOG.filter(
   267→    (r) =>
   268→      r.repo.toLowerCase().includes(q) ||
   269→      r.description.toLowerCase().includes(q) ||
   270→      r.family.includes(q) ||
   271→      r.builder.toLowerCase().includes(q) ||
   272→      r.tags.some((t) => t.includes(q)),
   273→  ).sort((a, b) => b.downloads - a.downloads);
   274→}

   299→interface SimState {

   301→  ticks: number;
   302→  stop: () => void;

   308→function uid(prefix = "id") {
   309→  return `${prefix}_${Math.random().toString(36).slice(2, 9)}`;
   310→}

   312→function nowTs() {
   313→  return Date.now();

   316→function emitLog(instanceId: string, kind: LogKind, text: string) {
   317→  const line: ConsoleLine = {
   318→    id: uid("log"),
   319→    instanceId,
   320→    ts: nowTs(),
   321→    kind,
   322→    text,
   323→  };
   324→  logSubscribers.forEach((fn) => fn(line));
   325→  return line;
   326→}

   332→function runStartupSequence(instance: LlamaInstance, store: LlamaStore) {
   333→  let cancelled = false;
   334→  let reqTimer: ReturnType<typeof setTimeout> | null = null;
   335→
   336→  const sim: SimState = {
   337→    running: true,
   338→    ticks: 0,
   339→    stop: () => {
   340→      cancelled = true;
   341→      sim.running = false;
   342→      if (reqTimer) clearTimeout(reqTimer);
   343→    },
   344→  };
   345→  instanceSims.set(instance.id, sim);
   346→
   347→  const steps: Array<[number, LogKind, string]> = [

   687→function startWatchdog() {
   688→  if (watchdogTimer) clearInterval(watchdogTimer);
   689→  watchdogTimer = setInterval(() => {
   690→    const s = useLlamaStore.getState();
   691→    const since = Date.now() - s.lastActivityAt;
   692→    const wsSettings = s.workspaceSettings[s.activeWorkspaceId] ?? defaultWorkspaceSettings;
   693→    const hibernateAfterMs = wsSettings.hibernateAfterSec * 1000;
   694→    const running = s.instances.filter((i) => i.status === "running" || i.status === "starting");
   695→
   696→    if (s.appStatus === "waking" || s.appStatus === "hibernating") return;
   697→
   698→    if (since >= hibernateAfterMs && running.length > 0 && s.appStatus !== "hibernating") {
   699→      s.setAppStatus("hibernating");
   700→      emitLog(SYSTEM_CONSOLE_ID, "warn", `[${fmtTime(new Date())}] [hibernate] idle ${Math.round(since / 1000)}s — unloading ${running.length} model(s) from VRAM`);
   701→      running.forEach((inst) => {
   702→        useLlamaStore.getState().stopInstance(inst.id);
   703→        useLlamaStore.setState((st) => ({
   704→          hibernatedInstanceIds: [...st.hibernatedInstanceIds, inst.id],
   705→          instances: st.instances.map((i) =>
   706→            i.id === inst.id
   707→              ? { ...i, hibernatedConfig: { name: i.name, model: i.model, profile: i.profile, port: i.port, host: i.host, gpu: i.gpu } }

   710→        }));
   711→      });
   712→      s.addNotification({ kind: "warn", title: "Hibernation started", body: `${running.length} model(s) unloaded from VRAM after ${Math.round(since / 1000)}s idle.` });

   719→    } else if (since < hibernateAfterMs * 0.6 && s.appStatus === "idle") {
   720→      s.setAppStatus("active");
   721→    }
   722→  }, 3000);

   725→let metricsTimer: ReturnType<typeof setInterval> | null = null;
   726→function startMetricsTicker() {

   729→    const s = useLlamaStore.getState();
   730→    const running = s.instances.filter((i) => i.status === "running");
   731→    const totalTps = running.reduce((sum, i) => sum + i.tokensPerSec, 0);
   732→    const totalMem = running.reduce((sum, i) => sum + i.memoryMb, 0);
   733→    const totalReq = running.reduce((sum, i) => sum + i.requestsPerMin, 0);
   734→    const isHibernating = s.appStatus === "hibernating";
   735→    const sample: MetricSample = {
   736→      t: Date.now(),
   737→      cpu: isHibernating ? 1 + Math.random() * 2 : 8 + Math.random() * 18 + running.length * 3,
   738→      ram: isHibernating ? 18 + Math.random() * 4 : 32 + Math.random() * 6 + totalMem / 1024,
   739→      gpu: isHibernating ? 0 : Math.min(100, 15 + running.length * 22 + Math.random() * 18),
   740→      gpuMem: isHibernating ? 0 : Math.min(100, (totalMem / 12288) * 100),
   741→      tps: totalTps,
   742→      reqPerMin: totalReq,
   743→    };
   744→    useLlamaStore.getState().pushMetric(sample);
   745→  }, 1500);
