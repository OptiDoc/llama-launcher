     1→/**
     2→ * LlamaLauncher state store + simulator.
     3→ *
     4→ * Client-side simulation of llama.cpp server instances. Each instance is a
     5→ * llama-server process that streams realistic startup logs to its own console
     6→ * buffer.
     7→ *
     8→ * Features:
     9→ * - Workspaces (isolated environments with own instances/profiles/releases)
    10→ * - App status / hibernation (auto-unload models after idle period,
    11→ *   hot-reload on next request) — hibernate delay is configurable
    12→ * - Profiles with scope: 'global' | 'model' (for sharing + auto-calibration)
    13→ * - HuggingFace model download (search-first, quantization picker)
    14→ * - llama.cpp release variants (cuda12 / cuda13 / vulkan / cpu / …)
    15→ * - Notifications (new GitHub release, download complete, …)
    16→ * - Real-time system metrics stream (CPU/RAM/GPU/tok/s) for the dashboard
    17→ * - Model metadata: builder, architecture, license, edit/delete/missing
    18→ */

   239→export function searchHFModels(query: string): HFSearchResult[] {
   240→  const q = query.trim().toLowerCase();
   241→  if (!q) return [];
   242→  return HF_CATALOG.filter(
   243→    (r) =>
   244→      r.repo.toLowerCase().includes(q) ||
   245→      r.description.toLowerCase().includes(q) ||
   246→      r.family.includes(q) ||
   247→      r.builder.toLowerCase().includes(q) ||
   248→      r.tags.some((t) => t.includes(q)),
   249→  ).sort((a, b) => b.downloads - a.downloads);
   250→}

   275→interface SimState {
   276→  running: boolean;
   277→  ticks: number;
   278→  stop: () => void;
   279→}
   280→
   281→const instanceSims = new Map<string, SimState>();
   282→const logSubscribers = new Set<(line: ConsoleLine) => void>();
   283→
   284→function uid(prefix = "id") {
   285→  return `${prefix}_${Math.random().toString(36).slice(2, 9)}`;
   286→}
   287→
   288→function nowTs() {
   289→  return Date.now();
   290→}

   292→function emitLog(instanceId: string, kind: LogKind, text: string) {
   293→  const line: ConsoleLine = {
   294→    id: uid("log"),
   295→    instanceId,
   296→    ts: nowTs(),
   297→    kind,
   298→    text,
   299→  };
   300→  logSubscribers.forEach((fn) => fn(line));
   301→  return line;
   302→}

   308→function runStartupSequence(instance: LlamaInstance, store: LlamaStore) {
   309→  let cancelled = false;
   310→  let reqTimer: ReturnType<typeof setTimeout> | null = null;
   311→
   312→  const sim: SimState = {
   313→    running: true,
   314→    ticks: 0,
   315→    stop: () => {
   316→      cancelled = true;
   317→      sim.running = false;
   318→      if (reqTimer) clearTimeout(reqTimer);
   319→    },
   320→  };
   321→  instanceSims.set(instance.id, sim);
   322→
   323→  const steps: Array<[number, LogKind, string]> = [
   324→    [50, "info", `$ llama-server --model ${instance.model} --host ${instance.host} --port ${instance.port} -c ${instance.ctxSize} -t ${instance.threads} -ngl ${instance.gpu === "cpu" ? 0 : 99}`],
   325→    [120, "debug", "build: 4402 (8f1f7e1) with AVX2=1 AVX512=1 BMI2=1 CUDA=1"],
   326→    [180, "info", "system_info: n_threads = " + instance.threads + " n_gpu_layers = 99"],
   327→    [260, "info", "loading model from " + instance.model],
   328→    [380, "debug", "llama_model_loader: loaded meta data with 24 KV pairs"],
   329→    [440, "debug", "llama_model_loader: - kv[ 0]:                       general.name = " + instance.name],
   330→    [520, "debug", "llama_model_loader: - kv[ 1]:          general.architecture = llama"],
   331→    [600, "info", `llama_model_loader: - kv[12]:                      llama.context_length = ${instance.ctxSize}`],
   332→    [680, "debug", "llama_model_loader: - type  f16:  221 tensors"],
   333→    [760, "info", "llama_model_loader: loading model part 1/1 from '" + instance.model + "'"],
   334→    [900, "info", "llama_model_loader: model size = " + (instance.model.includes("7b") ? "13.9" : "4.1") + " GiB"],
   335→    [980, "success", "llama_model_loader: model loaded successfully"],
   336→    [1060, "info", "using CUDA for GPU acceleration"],
   337→    [1140, "debug", "llama_kv_cache:  CUDA0 KV buffer size = " + Math.round(instance.ctxSize * 0.5) + " MiB"],
   338→    [1220, "info", "llama_new_context_with_model: n_ctx = " + instance.ctxSize + ", batch = 512"],
   339→    [1320, "success", "llama_new_context_with_model: graph initialized"],
   340→    [1400, "info", `llama server: listening on ${instance.host}:${instance.port}`],
   341→    [1480, "info", "llama server: loading model took " + (1200 + Math.floor(Math.random() * 600)) + " ms"],
   342→    [1560, "success", `llama server: model loaded, ready for requests at http://${instance.host}:${instance.port}`],
   343→  ];
   344→
   345→  let elapsed = 0;
   346→  steps.forEach(([delay, kind, text]) => {
   347→    if (cancelled) return;
   348→    setTimeout(() => {
   349→      if (cancelled) return;
   350→      emitLog(instance.id, kind, `[${fmtTime(new Date())}] ${text}`);
   351→    }, delay);
   352→    elapsed = delay;
   353→  });
   354→
   355→  setTimeout(() => {
   356→    if (cancelled) return;
   357→    store.markRunning(instance.id);
   358→    emitLog(instance.id, "success", `[${fmtTime(new Date())}] ✓ Server ready. Listening on http://${instance.host}:${instance.port}/completions`);
   359→
   360→    const tick = () => {
   361→      if (cancelled || !sim.running) return;
   362→      sim.ticks += 1;
   363→      const t = sim.ticks;
   364→      store.registerActivity();
   365→      if (t % 3 === 0) {
   366→        const promptTok = 8 + Math.floor(Math.random() * 80);

   645→function startWatchdog() {
   646→  if (watchdogTimer) clearInterval(watchdogTimer);
   647→  watchdogTimer = setInterval(() => {
   648→    const s = useLlamaStore.getState();
   649→    const since = Date.now() - s.lastActivityAt;
   650→    const wsSettings = s.workspaceSettings[s.activeWorkspaceId] ?? defaultWorkspaceSettings;
   651→    const hibernateAfterMs = wsSettings.hibernateAfterSec * 1000;
   652→    const running = s.instances.filter((i) => i.status === "running" || i.status === "starting");
   653→
   654→    if (s.appStatus === "waking" || s.appStatus === "hibernating") return;
   655→
   656→    if (since >= hibernateAfterMs && running.length > 0 && s.appStatus !== "hibernating") {
   657→      s.setAppStatus("hibernating");
   658→      emitLog(SYSTEM_CONSOLE_ID, "warn", `[${fmtTime(new Date())}] [hibernate] idle ${Math.round(since / 1000)}s — unloading ${running.length} model(s) from VRAM`);
   659→      running.forEach((inst) => {
   660→        useLlamaStore.getState().stopInstance(inst.id);
   661→        useLlamaStore.setState((st) => ({
   662→          hibernatedInstanceIds: [...st.hibernatedInstanceIds, inst.id],
   663→          instances: st.instances.map((i) =>
   664→            i.id === inst.id
   665→              ? { ...i, hibernatedConfig: { name: i.name, model: i.model, profile: i.profile, port: i.port, host: i.host, gpu: i.gpu } }
   666→              : i,
   667→          ),
   668→        }));
   669→      });
   670→      s.addNotification({ kind: "warn", title: "Hibernation started", body: `${running.length} model(s) unloaded from VRAM after ${Math.round(since / 1000)}s idle.` });
   671→      setTimeout(() => {
   672→        emitLog(SYSTEM_CONSOLE_ID, "info", `[${fmtTime(new Date())}] [hibernate] all models unloaded. VRAM freed. Awaiting next request to hot-reload.`);
   673→      }, 1200);
   674→    } else if (since >= hibernateAfterMs * 0.6 && s.appStatus === "active") {
   675→      s.setAppStatus("idle");
   676→      emitLog(SYSTEM_CONSOLE_ID, "debug", `[${fmtTime(new Date())}] [idle] no activity for ${Math.round(since / 1000)}s — will hibernate in ${Math.max(0, Math.round((hibernateAfterMs - since) / 1000))}s`);
   677→    } else if (since < hibernateAfterMs * 0.6 && s.appStatus === "idle") {
   678→      s.setAppStatus("active");
   679→    }
   680→  }, 3000);

  1072→export function uptimeString(startedAt?: number) {
  1073→  if (!startedAt) return "--";
  1074→  const sec = Math.floor((Date.now() - startedAt) / 1000);
  1075→  const h = Math.floor(sec / 3600);
  1076→  const m = Math.floor((sec % 3600) / 60);
  1077→  const s = sec % 60;
  1078→  if (h > 0) return `${h}h ${m}m`;
  1079→  if (m > 0) return `${m}m ${s}s`;
  1080→  return `${s}s`;
  1081→}

  1087→export function fmtNum(n: number) {
  1088→  if (n >= 1_000_000) return `${(n / 1_000_000).toFixed(1)}M`;
  1089→  if (n >= 1_000) return `${(n / 1_000).toFixed(1)}K`;
  1090→  return String(n);
  1091→}

  1093→export function fmtBytes(gb: number) {
  1094→  if (gb >= 1) return `${gb.toFixed(1)} GB`;
  1095→  return `${(gb * 1024).toFixed(0)} MB`;
  1096→}
